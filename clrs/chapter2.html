insertion_sort(A,n)
  for i = 2 to n
    key = A[i]
    // Insert A[i] into the sorted subarray A[1:i-1].
    j = i - 1
    while j > 0 and A[j] > key
      A[j + 1] = A[j]
      j = j - 1
    A[j + 1] = key

selection_sort(A,n)
  for i = 1 to n - 1
    // Find smallest element in A[i:n] and exchange with A[i].
    smallest = i
    for j = i + 1 to n
      if A[j] < A[smallest]
        smallest = j
    exchange A[i] with A[smallest]

bubble_sort(A,n)
  for i = 1 to n - 1
    for j = n downto i + 1
      if A[j] < A[j - 1]
        exchange A[j] with A[j - 1]
